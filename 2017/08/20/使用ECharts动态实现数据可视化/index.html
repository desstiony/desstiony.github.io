使用ECharts动态实现数据可视化

发表于 2017-08-20 | 分类于 web

  前段时间项目中需要用到动态的折线图来实现数据的可视化，于是谷歌一波发现百度有一个开源的ECharts。献上地址;官网有详细的介绍，此处不再累赘，结合自己的需求展示下代码。

效果图：


JS代码

var myChart = echarts.init(document.getElementById('myEchatrs'));
// 指定图表的配置项和数据
var option = {

    tooltip: {
        trigger: 'axis',//坐标轴触发提示框
    },

    legend: {    //图表上方的类别显示
        show: true,
        data: ['总云量', '厚云云量', '薄云云量']
    },

    color: [
        '#FF3333',    //总云量曲线颜色
        '#53FF53',    //厚云云量曲线颜色
        '#B15BFF',    //薄云云量颜色
    ],

    xAxis: {
        type: 'category',
        data: [],    //先设置数据值为空，后面用Ajax获取动态数据填入
    },

    yAxis: [
        {
            //第一个（左边）Y轴，yAxisIndex为0
            type: 'value',
            name: '云量',
            /* max: 120,
            min: -40, */
            axisLabel: {
                formatter: '{value}'    //控制输出格式
            }
        },
    ],

    series: [    //系列（内容）列表
        {
            name: '总云量',
            type: 'line',    //折线图
            symbol: 'emptycircle',    //设置折线图中表示每个坐标点的符号；emptycircle：空心圆；emptyrect：空心矩形；circle：实心圆；emptydiamond：菱形
            data: []
        },

        {
            name: '厚云云量',
            type: 'line',
            symbol: 'emptyrect',
            data: []
        },

        {
            name: '薄云云量',
            type: 'line',
            symbol: 'emptydiamond',
            data: []
        },
    ]
};

myChart.showLoading();    //数据加载完之前先显示一段简单的loading动画

var total = [];        //总云量数组（存放服务器返回的所有温度值）
var thick = [];        //厚云云量数组
var thin = [];        //薄云云量数组
var dates = [];        //时间数组
var cloudinfo;
var datas = [];

$.ajax({
    type: "post",
    async: true,
    url: "/user/showtoday",
    data: {},
    dataType: "json",
    contentType: "application/json",
    success: function (data) {
    cloudinfo = data.cloudinfos;

        if (data != null) {
            for (var i = 0; i < cloudinfo.length; i++) {
                total.push(cloudinfo[i].total_amount);
                thick.push(cloudinfo[i].thick_amount);
                thin.push(cloudinfo[i].thin_amount);
                dates.push(cloudinfo[i].time);
            }
            myChart.hideLoading();    //隐藏加载动画
            myChart.setOption({        //载入数据
                xAxis: {
                    data: dates    //填入X轴数据
                },
                series: [    //填入系列（内容）数据
                    {
                        name: '总云量',
                        data: total
                    },
                    {
                        name: '厚云云量',
                        data: thick
                    },
                    {
                        name: '薄云云量',
                        data: thin
                    },
                ]
            });
        }
        else {
            //when data is null
            alert("图表请求数据为空，可能服务器暂未录入近五天的观测数据，您可以稍后再试！");
            myChart.hideLoading();
        }
    },
    error: function (errorMsg) {
        alert("图表请求数据失败，可能是服务器开小差了");
        myChart.hideLoading();
    }
})
myChart.setOption(option);    //载入图表

后台代码

(ssm框架)此处只展示了controller的部分代码:

/**

@RequestMapping("/showtoday")
@ResponseBody
public User getTodayData(HttpServletRequest request){
    User user = (User)request.getSession().getAttribute("userinfo");
    int cid = user.getCid();
    User todaydata = userService.selectToday(cid);
    return todaydata;
}

总结

使用echarts框架，配合后台提供的接口，利用ajax动态请求后台获取数据，使用fastjson将数据转化为json数组返回。
